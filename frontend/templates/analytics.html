📊 Analytics & Queries

Aggregate · GROUP BY · SET Ops · Subqueries · JOINs
⚡ Disruptions by Type
SELECT disruption_type, COUNT(*) AS count, SUM(CASE WHEN resolution_status='Active' THEN 1 ELSE 0 END) AS active FROM disruption_event GROUP BY disruption_type;
✈ Flights by Airline — COUNT, SUM with CASE WHEN
SELECT al.airline_name, COUNT(f.flight_id) AS total_flights, SUM(CASE WHEN f.flight_status='Delayed' THEN 1 ELSE 0 END) AS delayed, SUM(CASE WHEN f.flight_status='Cancelled' THEN 1 ELSE 0 END) AS cancelled FROM flight f JOIN airline al ON f.airline_id=al.airline_id GROUP BY al.airline_id, al.airline_name;
💰 Fare Statistics — MIN · MAX · AVG · COUNT
SELECT f.flight_number, MIN(b.total_fare) AS min_fare, MAX(b.total_fare) AS max_fare, AVG(b.total_fare) AS avg_fare, COUNT(b.booking_id) AS bookings FROM booking b JOIN flight f ON b.flight_id=f.flight_id GROUP BY f.flight_id, f.flight_number;
🔗 Set Operations — UNION · INTERSECT · EXCEPT
-- UNION: fatigued pilots ∪ assigned pilots -- INTERSECT: fatigued AND currently assigned (SAFETY RISK!) -- EXCEPT: fatigued but NOT assigned (safe)
🔍 Subquery — Above-Average Surge
SELECT * FROM price_surge_log WHERE surge_percentage > ( SELECT AVG(surge_percentage) FROM price_surge_log ) ORDER BY surge_percentage DESC;
🔀 JOIN Types Showcase
INNER JOIN
SELECT f.flight_number, al.airline_name FROM flight f INNER JOIN airline al ON f.airline_id = al.airline_id;
Returns only flights that have a matching airline record.
LEFT JOIN
SELECT p.passenger_name, b.booking_id FROM passenger p LEFT JOIN booking b ON p.passenger_id = b.passenger_id;
Returns all passengers, including those without bookings.
MULTI-TABLE JOIN
SELECT f.flight_number, al.airline_name, src.city AS from, dst.city AS to FROM flight f JOIN airline al ON f.airline_id = al.airline_id JOIN airport src ON f.source_airport_id = src.airport_id JOIN airport dst ON f.destination_airport_id = dst.airport_id;
JOIN with Subquery
SELECT md.*, de.disruption_type FROM management_decision md JOIN disruption_event de ON md.disruption_id = de.disruption_id WHERE de.severity_level IN ( SELECT DISTINCT severity_level FROM disruption_event WHERE resolution_status = 'Active' );
